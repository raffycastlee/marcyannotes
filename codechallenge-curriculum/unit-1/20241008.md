const isAllLowerCase = (str) => {
  for (let i = 0; i < str.length; i++) {
    asciiCode = str.charCodeAt(i);
    if (asciiCode == 32) { continue; }
    if (asciiCode < 97 || asciiCode > 122) { return false; }
  }
  return true
}
